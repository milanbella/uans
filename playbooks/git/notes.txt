Set gaow.git repository on remote server:

> su - git
> cd repos
> mkdir  gaow.git
> cd gaow.git
> git init --bare

# On local server inside gaow project folder:

> cd gaow
> git init
> git add .
> git commit -m 'Initial commit'
> git remote add origin  git@212.24.97.13:repos/gaow.git
> git push origin -u main
